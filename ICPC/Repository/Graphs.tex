\vspace{-5pt}

\vspace{-5pt}
\begin{itemize}[noitemsep]
  \item u = lca(u, v), query(tin[u], tin[v]) 
  \item u $\neq$ lca(u, v), query(tout[u], tin[v]) + query(tin[$lca$], tin[$lca$])
\end{itemize}
